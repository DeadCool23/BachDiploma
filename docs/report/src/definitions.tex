\chapter*{ПЕРЕЧЕНЬ СОКРАЩЕНИЙ И ОБОЗНАЧЕНИЙ}
\addcontentsline{toc}{chapter}{ПЕРЕЧЕНЬ СОКРАЩЕНИЙ И ОБОЗНАЧЕНИЙ}

\noindent
\begin{description}[leftmargin=3cm, labelwidth=2.5cm, labelsep=0.5cm, align=left]
    \item[ВР] Высокое разрешение (High Resolution) --- изображение высокого разрешения
    \item[НР] Низкое разрешение (Low Resolution) --- изображение низкого разрешения
    \item[СР] Суперразрешение (Super-Resolution) --- обработка изображений низкого разрешение, направленная на получение изображений более высокого разрешения
    \item[CNN] Сверточная нейронная сеть (Convolutional Neural Network) --- класс нейронных сетей, разработанный для обработки изображений
    \item[EDSR]  Улучшенная глубокая сеть для суперразрешения (Enhanced Deep Super-Resolution) --- глубокая остаточная сеть, основанная на архитектуре SRResNet
    \item[GAN] Генеративно-состязательная сеть (Generative Adversarial Network) --- класс нейронных сетей, состоящий генератора и дискриминатора, для генерации фотореалистичных изображений
    \item[PSNR] Пиковое отношение сигнала к шуму (Peak Signal-to-Noise Ratio) --- метрика, измеряющая точность восстановления пикселей на основе среднеквадратичной ошибки
    \item[RCAN] Сеть с остаточным канальным вниманием (Residual Channel Attention Network) --- глубокая остаточная сеть с механизмом канального внимания, основанная на архитектуре EDSR
    \item[RGB] Красный, зеленый, синий (Red, Green, Blue) — цветовая модель
    \item[SRCNN] Сверточная нейронная сеть для суперразрешения (Super-Resolution Convolutional Neural Network) --- одна из первых глубоких моделей, доказавшая эффективность сверточных нейронных сетей при повышении разрешения изображений
    \item[SRGAN] Генеративно-состязательная сеть для суперразрешения (Super-Resolution Generative Adversarial Network) --- генеративно-состязательная нейронная сеть для повышения разрешения, генерирующая фотореалистичные текстуры, не ориентируясь на пиксельную точность
    \item[SSIM] Индекс структурного сходства (Structural Similarity Index) --- метрика, оценивающая сохранение структурной информации и воспринимаемого качества изображения
\end{description}
